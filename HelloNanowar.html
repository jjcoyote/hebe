package it.nanowar.ofsteel.helloworld;

public class HelloWorldMainLauncherClass {
    
    private static final boolean METAL = true;

    public static void main(String[] args) {
        System.out.println("🚀 Programmer start!");
        HelloWorldMainLauncherClass tizio = new HelloWorldMainLauncherClass(2);
        tizio.songRefrain();
        System.out.println("🛑 Program finished");
        if(METAL == false) {
            System.out.println("Hello World, but it's not metal enough!");
        }
    }

    private int foo;

    public HelloWorldMainLauncherClass(int foo) {
        super();
        this.foo = foo;
    }

    public void songRefrain() {
        chorus(); // Let's rock!
    }

    private void chorus() {
        for(int i = 0; i < foo; i++) {
            System.out.println("🎸 Hello World!");
        }

        int pippo = 0;
        while(pippo < foo) {
            System.out.println("🥁 Hello World, again!");
            pippo++;
        }

        introduceBug();
    }

    private void introduceBug() {
        String JoeyDeCaio = null;
        try {
            System.out.println(JoeyDeCaio.length());
        } catch(NullPointerException e) {
            System.out.println("🤘 NullPointerException! Now THIS is metal!");
        }
    }
}
